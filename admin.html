const { createClient } = require('@supabase/supabase-js'); const jwt = require('jsonwebtoken'); const supabase = createClient( process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY ); const ADMIN_SECRET = process.env.JWT_SECRET + '_admin'; const ADMIN_PASSWORDS = [ process.env.ADMIN_PASSWORD, process.env.ADMIN_PASSWORD_2, process.env.ADMIN_PASSWORD_3, ].filter(Boolean); function signAdminToken() { return jwt.sign({ admin: true }, ADMIN_SECRET, { expiresIn: '8h' }); } function verifyAdminToken(req) { const auth = req.headers.authorization || ''; const token = auth.replace('Bearer ', ''); if (!token) return false; try { const d = jwt.verify(token, ADMIN_SECRET); return d.admin === true; } catch(e) { return false; } } module.exports = async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization'); if (req.method === 'OPTIONS') return res.status(200).end(); if (req.method !== 'POST') return res.status(405).json({ error: 'Method not allowed' }); const { action } = req.body; // ── Login ───────────────────────────────────────────────────── if (action === 'login') { const { password } = req.body; if (!password || !ADMIN_PASSWORDS.includes(password)) { return res.status(401).json({ error: 'Contraseña incorrecta' }); } return res.json({ ok: true, token: signAdminToken() }); } if (!verifyAdminToken(req)) { return res.status(401).json({ error: 'No autorizado' }); } // ── Stats generales ─────────────────────────────────────────── if (action === 'stats') { const [users, payments, prizes, pity] = await Promise.all([ supabase.from('users').select('id, gems', { count: 'exact' }), supabase.from('payments').select('amount_mxn, gems_granted, status'), supabase.from('prizes').select('redeemed'), supabase.from('global_pity').select('pull_count').maybeSingle(), ]); const approvedPayments = (payments.data || []).filter(p => p.status === 'approved' || p.status === 'manual'); const revenue = approvedPayments.reduce((s, p) => s + (p.amount_mxn || 0), 0); const gemsTotal = approvedPayments.reduce((s, p) => s + (p.gems_granted || 0), 0); const activePrizes = (prizes.data || []).filter(p => !p.redeemed).length; const totalGems = (users.data || []).reduce((s, u) => s + (u.gems || 0), 0); return res.json({ ok: true, users: users.count || 0, payments: approvedPayments.length, revenue, gemsTotal, activePrizes, totalGems, globalPulls: pity.data?.pull_count || 0, }); } // ── Lista de usuarios ───────────────────────────────────────── if (action === 'users') { const { data, error } = await supabase .from('users') .select('id, email, username, gems, created_at, is_test') .order('created_at', { ascending: false }); if (error) return res.status(500).json({ error: error.message }); // Conteos rápidos por usuario — limit alto para no truncar const [collections, redemptions, prizes, payments] = await Promise.all([ supabase.from('collections').select('user_id, rarity').limit(100000), supabase.from('card_redemptions').select('user_id').limit(100000), supabase.from('prizes').select('user_id, redeemed').limit(100000), supabase.from('payments').select('user_id, amount_mxn, status').limit(100000), ]); const colMap = {}; (collections.data || []).forEach(c => { colMap[c.user_id] = (colMap[c.user_id] || 0) + 1; }); const redMap = {}; (redemptions.data || []).forEach(r => { redMap[r.user_id] = (redMap[r.user_id] || 0) + 1; }); const prizeMap = {}; (prizes.data || []).forEach(p => { prizeMap[p.user_id] = (prizeMap[p.user_id] || 0) + 1; }); const spendMap = {}; (payments.data || []).filter(p => p.status === 'approved' || p.status === 'manual').forEach(p => { spendMap[p.user_id] = (spendMap[p.user_id] || 0) + (p.amount_mxn || 0); }); const users = (data || []).map(u => ({ ...u, cards: colMap[u.id] || 0, redeemed: redMap[u.id] || 0, prizes: prizeMap[u.id] || 0, spent: spendMap[u.id] || 0, })); return res.json({ ok: true, users }); } // ── Detalle de un usuario ───────────────────────────────────── if (action === 'user_detail') { const { user_id } = req.body; if (!user_id) return res.status(400).json({ error: 'user_id requerido' }); const [user, allCollections, redemptions, allPrizes, payments] = await Promise.all([ supabase.from('users').select('*').eq('id', user_id).maybeSingle(), supabase.from('collections').select('user_id, rarity').limit(100000), supabase.from('card_redemptions').select('rarity, redeemed_at, email').eq('user_id', user_id).order('redeemed_at', { ascending: false }), supabase.from('prizes').select('user_id, name, code, redeemed, obtained_at').limit(100000), supabase.from('payments').select('amount_mxn, gems_granted, status, created_at').eq('user_id', user_id).order('created_at', { ascending: false }), ]); if (!user.data) return res.status(404).json({ error: 'Usuario no encontrado' }); // Construir colMap y prizeMap primero const colMap = {}; (allCollections.data || []).forEach(c => { if (!colMap[c.user_id]) colMap[c.user_id] = []; colMap[c.user_id].push(c); }); const prizeMap = {}; (allPrizes.data || []).forEach(p => { if (!prizeMap[p.user_id]) prizeMap[p.user_id] = []; prizeMap[p.user_id].push(p); }); // Intentar con el user_id directo y con String() como fallback const collection = colMap[user_id] || colMap[String(user_id)] || []; const prizes = prizeMap[user_id] || prizeMap[String(user_id)] || []; // Cartas por rareza const byRarity = { common:0, rare:0, epic:0, legendary:0, secret:0 }; collection.forEach(c => { byRarity[c.rarity] = (byRarity[c.rarity] || 0) + 1; }); const approvedPayments = (payments.data || []).filter(p => p.status === 'approved' || p.status === 'manual'); const totalSpent = approvedPayments.reduce((s, p) => s + (p.amount_mxn || 0), 0); const totalGemsBought = approvedPayments.reduce((s, p) => s + (p.gems_granted || 0), 0); return res.json({ ok: true, user: user.data, stats: { totalCards: collection.length, byRarity, totalRedeemed: (redemptions.data || []).length, totalPrizes: prizes.length, activePrizes: prizes.filter(p => !p.redeemed).length, totalSpent, totalGemsBought, totalPurchases: approvedPayments.length, }, prizes: prizes, redemptions: redemptions.data || [], payments: approvedPayments, }); } // ── Toggle cuenta test ──────────────────────────────────────── if (action === 'toggle_test') { const { user_id, is_test } = req.body; if (!user_id) return res.status(400).json({ error: 'user_id requerido' }); const { error } = await supabase .from('users').update({ is_test: !!is_test }).eq('id', user_id); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true }); } // ── Agregar Maná ────────────────────────────────────────────── if (action === 'add_gems') { const { user_id, amount } = req.body; if (!user_id || !amount || amount <= 0) return res.status(400).json({ error: 'Datos inválidos' }); const { data: user } = await supabase.from('users').select('gems').eq('id', user_id).maybeSingle(); if (!user) return res.status(404).json({ error: 'Usuario no encontrado' }); const newGems = user.gems + Math.floor(amount); const { error } = await supabase.from('users').update({ gems: newGems }).eq('id', user_id); if (error) return res.status(500).json({ error: error.message }); await supabase.from('payments').insert({ user_id, mp_payment_id: 'MANUAL_' + Date.now(), amount_mxn: 0, gems_granted: Math.floor(amount), status: 'manual' }); return res.json({ ok: true, gems: newGems }); } // ── Pagos ───────────────────────────────────────────────────── if (action === 'payments') { const { data, error } = await supabase .from('payments') .select('*, users(email, username)') .order('created_at', { ascending: false }) .limit(100); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true, payments: data }); } // ── Premios ─────────────────────────────────────────────────── if (action === 'prizes') { const { data, error } = await supabase .from('prizes') .select('id, name, code, redeemed, obtained_at, user_id, users(email, username)') .order('obtained_at', { ascending: false }); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true, prizes: data }); } // ── Canjear premio ──────────────────────────────────────────── if (action === 'redeem_prize') { const { prize_id } = req.body; if (!prize_id) return res.status(400).json({ error: 'ID requerido' }); const { error } = await supabase.from('prizes').update({ redeemed: true }).eq('id', prize_id); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true }); } // ── Canjes de cartas ────────────────────────────────────────── if (action === 'redemptions') { const { data, error } = await supabase .from('card_redemptions') .select('*') .order('redeemed_at', { ascending: false }) .limit(200); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true, redemptions: data }); } // ── Anuncios ────────────────────────────────────────────────── if (action === 'get_announcements') { const { data, error } = await supabase .from('announcements') .select('*') .order('created_at', { ascending: false }); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true, announcements: data }); } if (action === 'create_announcement') { const { title, message, expires_at } = req.body; if (!title || !message) return res.status(400).json({ error: 'Título y mensaje requeridos' }); const insert = { title, message, active: true }; if (expires_at) insert.expires_at = expires_at; const { data, error } = await supabase.from('announcements').insert(insert).select().single(); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true, announcement: data }); } if (action === 'toggle_announcement') { const { announcement_id, active } = req.body; if (!announcement_id) return res.status(400).json({ error: 'ID requerido' }); const { error } = await supabase .from('announcements').update({ active: !!active }).eq('id', announcement_id); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true }); } if (action === 'delete_announcement') { const { announcement_id } = req.body; if (!announcement_id) return res.status(400).json({ error: 'ID requerido' }); const { error } = await supabase.from('announcements').delete().eq('id', announcement_id); if (error) return res.status(500).json({ error: error.message }); return res.json({ ok: true }); } return res.status(400).json({ error: 'Acción inválida' }); };
